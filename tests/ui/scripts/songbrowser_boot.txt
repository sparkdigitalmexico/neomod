# pins the async songbrowser boot path: set_active_ui_screen songbrowser triggers the
# database load that a plain boot does not. osu_folder points at the empty fixture dir
# (run.sh creates it) so the carousel stays empty and machine-independent; the screen
# must come up active with its manually-managed containers reachable.
options_save_on_back 0
debug_fixed_frametime 0.0166667
ui_trace 1
mouse_raw_input 0
mouse_sensitivity 1
ui_scale_to_dpi 0
windowed_resolution 1280x720
mouse_to 10 10
@wait 30
osu_folder uitest_osu_folder
set_active_ui_screen songbrowser
@wait 300
ui_assert screen_active songbrowser 1
ui_assert visible songbrowser 1
ui_assert exists songbrowser_scores 1
ui_assert exists songbrowser_search 1
exit
